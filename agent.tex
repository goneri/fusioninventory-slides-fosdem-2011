\begin{frame}
    \frametitle{Wake on LAN}

    \begin{block}{What?}
    \begin{itemize}
        \item awake computer.
    \end{itemize}
    \end{block}

\pause

    \begin{block}{How?}
    send the Magic Packet from an agent in the same network
    \begin{itemize}
        \item send raw ethernet packet
        \item UDP packet still possible
    \end{itemize}
    \end{block}

\pause

    \begin{block}{Benefit}
    \begin{itemize}
        \item no firewall issue
        \item nor special routage rule needed
    \end{itemize}
    \end{block}

\end{frame}
